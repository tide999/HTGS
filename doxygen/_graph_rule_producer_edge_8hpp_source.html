37  void updateEdge(std::shared_ptr<Connector<T>> newConnector, AnyTaskGraphConf *graph) override {
38  auto taskManager = graph->getTaskManager(bookkeeper);
39  auto oldConnector = this->getGraphConnector();
40 
41  Bookkeeper<T> *tmBookkeeper = (Bookkeeper<T> *)taskManager->getTaskFunction();
42 
43  // Find the RuleManager that has the same out
44  std::list<AnyRuleManagerInOnly<T> *> *ruleManagers = tmBookkeeper->getRuleManagers();
45 
46  for (AnyRuleManagerInOnly<T> *ruleManager : *ruleManagers)
47  {
48  if (ruleManager->getConnector() == oldConnector)
49  {
50  ruleManager->setOutputConnector(newConnector);
51  }
52  }
53 
54  this->setConnector(newConnector);
55  }
